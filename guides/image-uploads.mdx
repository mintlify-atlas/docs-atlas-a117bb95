Novel includes built-in support for image uploads via drag-and-drop, paste, and the slash command. This guide shows you how to set up image uploads with various storage providers.

## How Image Uploads Work

Novel handles images through three methods:

1. **Paste** - Users can paste images from their clipboard
2. **Drag and Drop** - Users can drag images into the editor
3. **Slash Command** - Users can type `/image` to upload via file picker

All methods use the same upload handler.

## Quick Start with Vercel Blob

<Steps>

### Step 1: Install Vercel Blob

```bash
npm install @vercel/blob
```

### Step 2: Create Upload API Route

Create `/api/upload/route.ts`:

```ts
import { put } from "@vercel/blob";
import { NextResponse } from "next/server";

export const runtime = "edge";

export async function POST(req: Request) {
  if (!process.env.BLOB_READ_WRITE_TOKEN) {
    return new Response(
      "Missing BLOB_READ_WRITE_TOKEN. Don't forget to add that to your .env file.",
      {
        status: 401,
      },
    );
  }

  const file = req.body || "";
  const filename = req.headers.get("x-vercel-filename") || "file.txt";
  const contentType = req.headers.get("content-type") || "text/plain";
  const fileType = `.${contentType.split("/")[1]}`;

  // construct final filename based on content-type if not provided
  const finalName = filename.includes(fileType)
    ? filename
    : `${filename}${fileType}`;
  const blob = await put(finalName, file, {
    contentType,
    access: "public",
  });

  return NextResponse.json(blob);
}
```

### Step 3: Add Environment Variable

Add to `.env.local`:

```bash
BLOB_READ_WRITE_TOKEN=vercel_blob_rw_...
```

### Step 4: Use the Editor

Novel automatically uses `/api/upload` endpoint:

```tsx
import { Editor } from "novel";

export default function MyEditor() {
  return <Editor />;
}
```

</Steps>

## Upload Handler Implementation

Here's how Novel's upload handler works:

```tsx
export const handleImageUpload = (file: File) => {
  return new Promise((resolve) => {
    toast.promise(
      fetch("/api/upload", {
        method: "POST",
        headers: {
          "content-type": file?.type || "application/octet-stream",
          "x-vercel-filename": file?.name || "image.png",
        },
        body: file,
      }).then(async (res) => {
        // Successfully uploaded image
        if (res.status === 200) {
          const { url } = (await res.json()) as BlobResult;
          // preload the image
          let image = new Image();
          image.src = url;
          image.onload = () => {
            resolve(url);
          };
          // No blob store configured
        } else if (res.status === 401) {
          resolve(file);
          throw new Error(
            "`BLOB_READ_WRITE_TOKEN` environment variable not found, reading image locally instead."
          );
          // Unknown error
        } else {
          throw new Error(`Error uploading image. Please try again.`);
        }
      }),
      {
        loading: "Uploading image...",
        success: "Image uploaded successfully.",
        error: (e) => e.message,
      }
    );
  });
};
```

## Image Upload Plugin

Novel uses a ProseMirror plugin to handle the upload flow:

```tsx
export function startImageUpload(file: File, view: EditorView, pos: number) {
  // check if the file is an image
  if (!file.type.includes("image/")) {
    toast.error("File type not supported.");
    return;

    // check if the file size is less than 20MB
  } else if (file.size / 1024 / 1024 > 20) {
    toast.error("File size too big (max 20MB).");
    return;
  }

  // A fresh object to act as the ID for this upload
  const id = {};

  // Replace the selection with a placeholder
  const tr = view.state.tr;
  if (!tr.selection.empty) tr.deleteSelection();

  const reader = new FileReader();
  reader.readAsDataURL(file);
  reader.onload = () => {
    tr.setMeta(uploadKey, {
      add: {
        id,
        pos,
        src: reader.result,
      },
    });
    view.dispatch(tr);
  };

  handleImageUpload(file).then((src) => {
    const { schema } = view.state;

    let pos = findPlaceholder(view.state, id);
    // If the content around the placeholder has been deleted, drop the image
    if (pos == null) return;

    // When BLOB_READ_WRITE_TOKEN is not valid or unavailable, read the image locally
    const imageSrc = typeof src === "object" ? reader.result : src;

    const node = schema.nodes.image.create({ src: imageSrc });
    const transaction = view.state.tr
      .replaceWith(pos, pos, node)
      .setMeta(uploadKey, { remove: { id } });
    view.dispatch(transaction);
  });
}
```

<Note>
The upload plugin shows a loading placeholder while the image uploads, then replaces it with the final image once the upload completes.
</Note>

    ```ts
            {
              folder: "novel-uploads",
              resource_type: "auto",
            },
            (error, result) => {
              if (error) reject(error);
              else resolve(result);
            }

      });
    ```

## Using Supabase Storage

<Tabs>
  <Tab title="Setup">
    ```bash
    npm install @supabase/supabase-js
    ```
  </Tab>

  <Tab title="API Route">
    ```ts
    import { createClient } from "@supabase/supabase-js";
    import { NextResponse } from "next/server";

    const supabase = createClient(
      process.env.NEXT_PUBLIC_SUPABASE_URL!,
      process.env.SUPABASE_SERVICE_ROLE_KEY!
    );

    export async function POST(req: Request) {
      const file = await req.arrayBuffer();
      const filename = req.headers.get("x-vercel-filename") || "image.png";
      const contentType = req.headers.get("content-type") || "image/png";

      const fileName = `${Date.now()}-${filename}`;

      const { data, error } = await supabase.storage
        .from("images")
        .upload(fileName, Buffer.from(file), {
          contentType,
          upsert: false,
        });

      if (error) {
        return new Response(error.message, { status: 500 });
      }

      const {
        data: { publicUrl },
      } = supabase.storage.from("images").getPublicUrl(data.path);

      return NextResponse.json({ url: publicUrl });
    }
    ```

## File Validation

Novel includes basic file validation. Here's how to customize it:

```tsx
export function startImageUpload(file: File, view: EditorView, pos: number) {
  // Custom file type validation
  const allowedTypes = ["image/jpeg", "image/png", "image/gif", "image/webp"];
  if (!allowedTypes.includes(file.type)) {
    toast.error("Only JPEG, PNG, GIF, and WebP images are allowed.");
    return;
  }

  // Custom file size validation (5MB limit)
  const maxSize = 5 * 1024 * 1024; // 5MB in bytes
  if (file.size > maxSize) {
    toast.error("File size must be less than 5MB.");
    return;
  }

  // ... rest of upload logic
}
```

## Image Compression

Here's how to compress images before upload:

```tsx
import imageCompression from "browser-image-compression";

export async function compressAndUpload(file: File) {
  try {
    const options = {
      maxSizeMB: 1,
      maxWidthOrHeight: 1920,
      useWebWorker: true,
    };

    const compressedFile = await imageCompression(file, options);

    // Upload compressed file
    const response = await fetch("/api/upload", {
      method: "POST",
      headers: {
        "content-type": compressedFile.type,
        "x-vercel-filename": compressedFile.name,
      },
      body: compressedFile,
    });

    return response.json();
  } catch (error) {
    console.error("Compression failed:", error);
    throw error;
  }
}
```

## Custom Upload Handler

To use a completely custom upload handler, you'll need to modify Novel's source:

<Steps>

### Step 1: Create custom upload function

```tsx
// lib/upload-handler.ts
export async function customImageUpload(file: File): Promise<string> {
  const formData = new FormData();
  formData.append("file", file);

  const response = await fetch("https://your-custom-api.com/upload", {
    method: "POST",
    body: formData,
  });

  const data = await response.json();
  return data.imageUrl;
}
```

### Step 2: Import and use in your editor

```tsx
import { Editor } from "novel";
import { customImageUpload } from "./lib/upload-handler";

// You'll need to fork Novel to use a custom upload handler
// or submit a PR to make the upload handler configurable
```

</Steps>

<Warning>
Currently, Novel's upload handler is not configurable via props. You'll need to either:
1. Use the `/api/upload` endpoint
2. Fork the Novel package and modify the upload handler
3. Submit a PR to make it configurable
</Warning>

## Image Paste Handler

Novel's default editor props include a paste handler:

```ts
handlePaste: (view, event) => {
  if (
    event.clipboardData &&
    event.clipboardData.files &&
    event.clipboardData.files[0]
  ) {
    event.preventDefault();
    const file = event.clipboardData.files[0];
    const pos = view.state.selection.from;

    startImageUpload(file, view, pos);
    return true;
  }
  return false;
},
```

## Image Drop Handler

Novel's default editor props include a drop handler:

```ts
handleDrop: (view, event, _slice, moved) => {
  if (
    !moved &&
    event.dataTransfer &&
    event.dataTransfer.files &&
    event.dataTransfer.files[0]
  ) {
    event.preventDefault();
    const file = event.dataTransfer.files[0];
    const coordinates = view.posAtCoords({
      left: event.clientX,
      top: event.clientY,
    });
    // here we deduct 1 from the pos or else the image will create an extra node
    startImageUpload(file, view, coordinates?.pos || 0 - 1);
    return true;
  }
  return false;
},
```

## Error Handling

Handle upload errors gracefully:

```tsx
export async function handleImageUpload(file: File) {
  try {
    const response = await fetch("/api/upload", {
      method: "POST",
      headers: {
        "content-type": file?.type || "application/octet-stream",
        "x-vercel-filename": file?.name || "image.png",
      },
      body: file,
    });

    if (!response.ok) {
      throw new Error(`Upload failed: ${response.statusText}`);
    }

    const { url } = await response.json();
    return url;
  } catch (error) {
    console.error("Image upload error:", error);
    toast.error("Failed to upload image. Please try again.");
    throw error;
  }
}
```

<Tip>
Consider implementing retry logic for failed uploads and showing progress indicators for large files.
</Tip>

## Complete Example

Here's a complete example with multiple storage options:

```ts
// app/api/upload/route.ts
import { put } from "@vercel/blob";
import { NextResponse } from "next/server";

export const runtime = "edge";

export async function POST(req: Request) {
  try {
    // Validate auth if needed
    const session = await getSession(req);
    if (!session) {
      return new Response("Unauthorized", { status: 401 });
    }

    const file = req.body || "";
    const filename = req.headers.get("x-vercel-filename") || "image.png";
    const contentType = req.headers.get("content-type") || "image/png";

    // Validate file type
    if (!contentType.startsWith("image/")) {
      return new Response("Only images are allowed", { status: 400 });
    }

    // Upload to Vercel Blob
    const blob = await put(`uploads/${Date.now()}-${filename}`, file, {
      contentType,
      access: "public",
    });

    // Log upload for analytics
    await logUpload({
      userId: session.userId,
      filename,
      url: blob.url,
      size: blob.size,
    });

    return NextResponse.json({ url: blob.url });
  } catch (error) {
    console.error("Upload error:", error);
    return new Response("Upload failed", { status: 500 });
  }
}
```
